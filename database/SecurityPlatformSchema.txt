create table employee
  (employee_ID 	serial,
   name			varchar(20),
   salt			numeric(8,0),
   hash_value	varchar(64),
   primary key (employee_ID)
  );
 
create table role
  (role_ID		serial,
   role_name	varchar(20) unique,
   primary key (role_ID)
  );

create table role_right
  (role_name	varchar(20),
   object		varchar(20),
   access_right	varchar(20),
   primary key (role_name, object, access_right),
   foreign key (role_name) references role(role_name)
  );
  
create table criteria 
  (criteria_ID		serial,
   criteria_name	varchar(20),
   comment			varchar(100),
   primary key	(criteria_ID)
  );
  
create table evidence
	(criteria_ID	serial,
	 evidence_ID	serial unique,
	 tag			varchar(20),
	 create_date	date,
	 last_edit_time	timestamp,
	 comment		varchar(100),
	 primary key (criteria_ID, evidence_ID),
	 foreign key (criteria_ID) references criteria(criteria_ID),
	 foreign key (evidence_ID) references evidence(evidence_ID)
	);

create table backup_evidence as
	(select * from evidence);

create table assign
	(employee_ID	serial,
	 role_ID		serial,
	 primary key (employee_ID, role_ID),
	 foreign key (employee_ID) references employee(employee_ID),
	 foreign key (role_ID) references role(role_ID)
	);

create table manage_criteria
	(role_ID		serial,
	 criteria_ID	serial,
	 primary key (role_ID, criteria_ID),
	 foreign key (criteria_ID) references criteria(criteria_ID),
	 foreign key (role_ID) references role(role_ID)	 
	);

create table manage_evidence
	(role_ID		serial,
	 evidence_ID	serial,
	 primary key (role_ID, evidence_ID),
	 foreign key (evidence_ID) references evidence(evidence_ID),
	 foreign key (role_ID) references role(role_ID)
	);
